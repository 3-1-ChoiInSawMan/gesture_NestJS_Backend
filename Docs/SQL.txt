CREATE TABLE `users` (
    `idx`            BIGINT       NOT NULL AUTO_INCREMENT,
    `email`          VARCHAR(255) NOT NULL,
    `nickname`       VARCHAR(500) NOT NULL,
    `id`             VARCHAR(200) NOT NULL,
    `password`       TEXT         NULL,
    `status_message` TEXT         NULL,
    `created_at`     DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at`     DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    `provider`       VARCHAR(200) NULL,
    `is_deactivated` BOOLEAN      NOT NULL DEFAULT FALSE,
    PRIMARY KEY (`idx`),
    UNIQUE KEY `users_email_key` (`email`),
    UNIQUE KEY `users_id_key` (`id`)
);

CREATE TABLE `chat_rooms` (
    `idx`        BIGINT        NOT NULL AUTO_INCREMENT,
    `name`       VARCHAR(1000) NOT NULL,
    `created_at` DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`idx`)
);

CREATE TABLE `chat_notices` (
    `idx`        BIGINT   NOT NULL AUTO_INCREMENT,
    `created_by` BIGINT   NOT NULL,
    `chat_idx`   BIGINT   NOT NULL,
    `content`    TEXT     NOT NULL,
    `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`created_by`) REFERENCES `users` (`idx`),
    FOREIGN KEY (`chat_idx`)   REFERENCES `chat_rooms` (`idx`)
);

CREATE TABLE `notification_settings` (
    `idx`     BIGINT       NOT NULL AUTO_INCREMENT,
    `user_id` BIGINT       NOT NULL,
    `type`    VARCHAR(100) NOT NULL,
    `is_used` BOOLEAN      NOT NULL DEFAULT TRUE,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`user_id`) REFERENCES `users` (`idx`)
);

CREATE TABLE `verification_codes` (
    `idx`        BIGINT      NOT NULL AUTO_INCREMENT,
    `email`      VARCHAR(255) NOT NULL,
    `code`       VARCHAR(50)  NOT NULL,
    `created_at` DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `expires_at` DATETIME     NOT NULL,
    `is_used`    BOOLEAN      NOT NULL DEFAULT FALSE,
    PRIMARY KEY (`idx`)
);

CREATE TABLE `quick_actions` (
    `idx`        BIGINT       NOT NULL AUTO_INCREMENT,
    `title`      VARCHAR(200) NOT NULL,
    `emoji_url`  VARCHAR(200) NOT NULL,
    `created_at` DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`idx`)
);

CREATE TABLE `quick_slot_presets` (
    `idx`         BIGINT NOT NULL AUTO_INCREMENT,
    `user_idx`    BIGINT NOT NULL,
    `action_id_1` BIGINT NULL,
    `action_id_2` BIGINT NULL,
    `action_id_3` BIGINT NULL,
    `action_id_4` BIGINT NULL,
    `action_id_5` BIGINT NULL,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`user_idx`)    REFERENCES `users`         (`idx`),
    FOREIGN KEY (`action_id_1`) REFERENCES `quick_actions` (`idx`),
    FOREIGN KEY (`action_id_2`) REFERENCES `quick_actions` (`idx`),
    FOREIGN KEY (`action_id_3`) REFERENCES `quick_actions` (`idx`),
    FOREIGN KEY (`action_id_4`) REFERENCES `quick_actions` (`idx`),
    FOREIGN KEY (`action_id_5`) REFERENCES `quick_actions` (`idx`)
);

CREATE TABLE `rooms` (
    `idx`             BIGINT       NOT NULL AUTO_INCREMENT,
    `host_idx`        BIGINT       NOT NULL,
    `title`           TEXT         NOT NULL,
    `description`     TEXT         NULL,
    `max_participant` INT          NOT NULL,
    `is_public`       BOOLEAN      NOT NULL DEFAULT TRUE,
    `password`        VARCHAR(255) NULL,
    `thumbnail_url`   VARCHAR(255) NULL,
    `created_at`      DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at`      DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`host_idx`) REFERENCES `users` (`idx`)
);

CREATE TABLE `room_members` (
    `idx`                  BIGINT       NOT NULL AUTO_INCREMENT,
    `room_idx`             BIGINT       NOT NULL,
    `user_idx`             BIGINT       NOT NULL,
    `role`                 VARCHAR(100) NOT NULL,
    `last_read_message_id` BIGINT       NULL,
    `created_at`           DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`room_idx`) REFERENCES `rooms` (`idx`),
    FOREIGN KEY (`user_idx`) REFERENCES `users` (`idx`)
);

CREATE TABLE `notifications` (
    `idx`        BIGINT       NOT NULL AUTO_INCREMENT,
    `user_idx`   BIGINT       NOT NULL,
    `target_id`  VARCHAR(100) NULL,
    `type`       VARCHAR(100) NOT NULL,
    `is_read`    BOOLEAN      NOT NULL,
    `content`    TEXT         NOT NULL,
    `created_at` DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`user_idx`) REFERENCES `users` (`idx`)
);

CREATE TABLE `friendships` (
    `idx`        BIGINT   NOT NULL AUTO_INCREMENT,
    `user_idx`   BIGINT   NOT NULL,
    `friend_idx` BIGINT   NOT NULL,
    `accept`     BOOLEAN  NOT NULL DEFAULT FALSE,
    `not_accept` BOOLEAN  NOT NULL DEFAULT FALSE,
    `request_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`user_idx`)   REFERENCES `users` (`idx`),
    FOREIGN KEY (`friend_idx`) REFERENCES `users` (`idx`)
);

CREATE TABLE `calls` (
    `idx`          BIGINT   NOT NULL AUTO_INCREMENT,
    `started_by`   BIGINT   NOT NULL,
    `is_recording` BOOLEAN  NOT NULL DEFAULT FALSE,
    `started_at`   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `ended_at`     DATETIME NULL,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`started_by`) REFERENCES `users` (`idx`)
);

CREATE TABLE `call_participants` (
    `idx`        BIGINT   NOT NULL AUTO_INCREMENT,
    `user_id`    BIGINT   NOT NULL,
    `calls_idx`  BIGINT   NOT NULL,
    `started_by` BIGINT   NOT NULL,
    `joined_at`  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `left_at`    DATETIME NULL,
    `field`      BOOLEAN  NOT NULL DEFAULT FALSE,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`user_id`)   REFERENCES `users` (`idx`),
    FOREIGN KEY (`calls_idx`) REFERENCES `calls` (`idx`)
);

CREATE TABLE `meetings` (
    `idx`        BIGINT       NOT NULL AUTO_INCREMENT,
    `created_by` BIGINT       NOT NULL,
    `title`      VARCHAR(100) NULL,
    `content`    TEXT         NOT NULL,
    `created_at` DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`created_by`) REFERENCES `users` (`idx`)
);

CREATE TABLE `refresh_tokens` (
    `idx`        BIGINT       NOT NULL AUTO_INCREMENT,
    `user_idx`   BIGINT       NOT NULL,
    `token`      TEXT         NOT NULL,
    `expires_at` DATETIME     NOT NULL,
    PRIMARY KEY (`idx`),
    UNIQUE KEY `refresh_tokens_user_idx_key` (`user_idx`),
    FOREIGN KEY (`user_idx`) REFERENCES `users` (`idx`)
);

CREATE TABLE `media` (
    `idx`      BIGINT       NOT NULL AUTO_INCREMENT,
    `user_idx` BIGINT       NOT NULL,
    `file`     TEXT         NULL,
    `url`      VARCHAR(36)  NOT NULL DEFAULT (UUID()),
    `name`     VARCHAR(140) NOT NULL,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`user_idx`) REFERENCES `users` (`idx`)
);

CREATE TABLE `chat_participants` (
    `idx`      BIGINT NOT NULL AUTO_INCREMENT,
    `chat_idx` BIGINT NOT NULL,
    `user_idx` BIGINT NULL,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`chat_idx`) REFERENCES `chat_rooms` (`idx`),
    FOREIGN KEY (`user_idx`) REFERENCES `users`      (`idx`)
);

CREATE TABLE `chat_messages` (
    `idx`        BIGINT                NOT NULL AUTO_INCREMENT,
    `sender_idx` BIGINT                NOT NULL,
    `chat_idx`   BIGINT                NOT NULL,
    `file_url`   BIGINT                NULL,
    `user_idx`   BIGINT                NULL,
    `target_idx` BIGINT                NULL,
    `message`    TEXT                  NULL,
    `is_deleted` BOOLEAN               NOT NULL DEFAULT FALSE,
    `created_at` DATETIME              NOT NULL DEFAULT CURRENT_TIMESTAMP,
    `updated_at` DATETIME              NULL,
    `type`       ENUM('text', 'file')  NOT NULL,
    PRIMARY KEY (`idx`),
    FOREIGN KEY (`sender_idx`) REFERENCES `users`      (`idx`),
    FOREIGN KEY (`chat_idx`)   REFERENCES `chat_rooms` (`idx`),
    FOREIGN KEY (`file_url`)   REFERENCES `media`      (`idx`)
);
